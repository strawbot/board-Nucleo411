Words // PWM / muscle wire
	setpwm			MW_CLI_SetPWM	    // ( n ) set muscle wire PWM duty to n percent (0-70)
	show-pwm		MW_CLI_ShowPWM	    // show PWM duty, supply/node voltages, wire resistance and contraction
	start-profile	MW_CLI_StartProfile // ( n ) apply n% PWM and begin 25.6 s resistance profile capture
	dump-profile	MW_DumpProfile	    // emit captured profile as CSV to CLI output
	cal-wire		MW_CLI_Calibrate    // run two-point self-calibration: R_max at 0% then R_min at 70%
	show-cal		MW_CLI_ShowCal      // show calibration state, R_max, R_min and measured travel span
	set-per			MW_CLI_SetPercent	//  ( n ) set closed-loop contraction target to n percent (0-100) and enable controller
	show-cl			MW_CLI_ShowCL		//  show closed-loop controller state: target, measured, error, rate, PWM
	cl-off			MW_CLI_CLOff		//  disable closed-loop controller; PWM holds at last value
	char-wire		MW_CLI_CharWire		//  ( n ) run 3-phase step-response test: pre-cool, heat at n%, cool to rest
	dump-char		MW_CLI_DumpChar		//  emit char-wire step-response data as CSV (paste into spreadsheet)
End

Words // system
	show-sys		show_sys		// show system info: clock frequencies and uptime
	show-time		show_timer		// show delta timer state and UTC tick counter
	show-timers		show_timers		// dump TIM1-TIM14 and RTC: clock gate, CEN, direction, PSC, ARR, CNT, active CC channels
	reboot			do_reboot		// reboot the device via NVIC system reset
	show-cli		show_cli		// display cli status
	pins			gpio_dump_all	// show all pins and states
End
